 CREATE TABLE MCAPROJECT.USERS_TYPE
   (	TYPE_ID VARCHAR2(4), 
	TYPE VARCHAR2(10) CONSTRAINT TYPE_NN NOT NULL ENABLE, 
CONSTRAINT TYPE_ID_PK PRIMARY KEY (TYPE_ID)
	 );

 CREATE TABLE MCAPROJECT.USER_DETAILS
   (	USER_ID VARCHAR2(7), 
   TYPE_ID VARCHAR2(4),
   USER_NAME VARCHAR2(10) CONSTRAINT USER_NAME_NN NOT NULL ENABLE, 
   TITLE VARCHAR2(5) CONSTRAINT TITLE_NN NOT NULL ENABLE, 
   FIRST_NAME VARCHAR2(15) CONSTRAINT FIRST_NAME_NN NOT NULL ENABLE, 
   LAST_NAME VARCHAR2(15) CONSTRAINT LAST_NAME_NN NOT NULL ENABLE, 
   EMAIL VARCHAR2(30) CONSTRAINT EMAIL_NN NOT NULL ENABLE, 
   PASSWORD VARCHAR2(100), 
   DOB DATE,
   ADDRESS VARCHAR2(100), 
   REGISTERED_TIME DATE,
   CONTACTNO NUMBER(10),
   SECRET_QUESTION VARCHAR2(100) CONSTRAINT SECRET_QUESTION_NN NOT NULL ENABLE,
   SECRET_ANSWER VARCHAR2(50) CONSTRAINT SECRET_ANSWER_NN NOT NULL ENABLE,
   CONSTRAINT USER_ID_PK PRIMARY KEY (USER_ID),
   CONSTRAINT TYPE_ID_FK FOREIGN KEY (TYPE_ID) REFERENCES MCAPROJECT.USERS_TYPE (TYPE_ID)
	 );

  CREATE TABLE MCAPROJECT.FACULTY
   (
   PROF_ID VARCHAR2(7), 
   PROF_NAME VARCHAR2(10),
   SUBJECT_ID NUMBER(4),
   CONSTRAINT PROF_ID_PK PRIMARY KEY (PROF_ID)
	 );
 
  CREATE TABLE MCAPROJECT.SUBJECT
   (	SUBJECT_ID NUMBER(4), 
	MODULE_ID VARCHAR2(10), 
	SUBJECT VARCHAR2(50), 
	MODULE VARCHAR2(50),
  UPLOADED_NOTES_FILE_NAME VARCHAR2(50),
  CONSTRAINT MODULE_ID_PK PRIMARY KEY (MODULE_ID),
   
   );
   
 CREATE TABLE MCAPROJECT.STUDENT_PROGRESS
   (
   STUDENT_ID VARCHAR2(7),
   MODULE_ID VARCHAR2(10),
   
   CONSTRAINT STUDENT_MODULE_IDS_PK PRIMARY KEY (STUDENT_ID, MODULE_ID),
   CONSTRAINT STUDENT_ID_FK FOREIGN KEY (STUDENT_ID) REFERENCES MCAPROJECT.USER_DETAILS (USER_ID),
   CONSTRAINT MODULE_ID_FK FOREIGN KEY (MODULE_ID) REFERENCES MCAPROJECT.SUBJECT (MODULE_ID)
	 );

 CREATE TABLE MCAPROJECT.NEWS_FEEDS
   (
   NEWS_ID VARCHAR2(5),
   PROF_ID VARCHAR2(7),
   NEWS VARCHAR2(4000),
   CONSTRAINT NEWS_ID_PK PRIMARY KEY (NEWS_ID),
   CONSTRAINT PROF_ID_NEWS_FEEDS_FK FOREIGN KEY (PROF_ID) REFERENCES MCAPROJECT.USER_DETAILS(USER_ID)
	 );

CREATE TABLE MCAPROJECT.module_notes
   (
   notes_ID VARCHAR2(7),
   PROF_ID VARCHAR2(7),
   MODULE_ID VARCHAR2(10),
   notes_fileName VARCHAR2(30),
   CONSTRAINT notes_ID_PK PRIMARY KEY (notes_ID),
   CONSTRAINT PROF_ID_SUBJECT_TEST_FK FOREIGN KEY (PROF_ID) REFERENCES MCAPROJECT.USER_DETAILS(USER_ID),
   CONSTRAINT MODULE_ID_SUBJECT_TEST_FK FOREIGN KEY (MODULE_ID) REFERENCES MCAPROJECT.SUBJECT(MODULE_ID)
	 );

CREATE TABLE "MCAPROJECT"."DISCUSSION_FORUM_PARTICIPANTS" 
   (	"USER_ID" VARCHAR2(7 BYTE), 
	"FULL_NAME" VARCHAR2(50 BYTE)
   );
